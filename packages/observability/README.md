      'order.id': orderId,
      'order.status': 'pending',
    },
  });

  try {
    // Tu l√≥gica aqu√≠
    const result = await doSomething();

    return result;
  } catch (error) {
    span.recordException(error);

      message: error.message,
    });
    throw error;
  } finally {
    span.end();
  }
}
